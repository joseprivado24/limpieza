async function startTask(id){

  const now=new Date().toISOString();

  const {error}=await db.from('tasks').update({started_at:now,status:'En curso'}).eq('id',id);

  if(error){toast('Error al iniciar','error');return;}

  toast('¡Tarea iniciada! ▶','orange');

  const t=Ts.find(x=>x.id===id);if(t){t.started_at=now;t.status='En curso';}

  viewTask(id);

}

async function finishTask(id){

  const now=new Date().toISOString();

  const t=Ts.find(x=>x.id===id);if(!t)return;

  const mins=calcMins(t.started_at,now);

  const allDone=t.items&&t.items.every(i=>i.done);

  const newStatus=allDone?'Completada':'En progreso';

  const {error}=await db.from('tasks').update({finished_at:now,duration_minutes:mins,status:newStatus}).eq('id',id);

  if(error){toast('Error al finalizar','error');return;}

  toast(`Tarea finalizada en ${fmtDur(mins)} ✓`,'success');

  t.finished_at=now;t.duration_minutes=mins;t.status=newStatus;

  if(timerInterval)clearInterval(timerInterval);timerInterval=null;

  viewTask(id);

  if(CU.role==='worker')renderWorkerView();

}
